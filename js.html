JavaScript Learning Notes

1. Data Types

JavaScript has several data types:

  • String, Number, Boolean
  • Null, Undefined
  • Object, Array

let name = "Alice";
let age = 25;
let isStudent = true;
let car = null;
let city;
let user = {name: "Bob", age: 30};
let colors = ["red", "green"];

2. Variables

Variables store data. Use let or const.


let x = 10;
const pi = 3.14;

3. Functions (Types)

1. Function Declaration

function greet(name) {
  return "Hello " + name;
}
console.log(greet("Alice"));
2. Function Expression

const sayHi = function(name) {
  return "Hi " + name;
};
console.log(sayHi("Bob"));
3. Arrow Function

const add = (a, b) => a + b;
console.log(add(3, 4));
4. Anonymous Function

setTimeout(function() {
  console.log("Anonymous function called!");
}, 1000);
5. Callback Function

function processInput(callback) {
  let name = "John";
  callback(name);
}

processInput(function(name) {
  console.log("Hello " + name);
});
6. IIFE (Immediately Invoked Function Expression)

(function() {
  console.log("IIFE executed immediately!");
})();

4. Arrays & Methods

Arrays are used to store multiple values in one variable. JavaScript provides many built-in methods to work with arrays.

Basic Array

// Declare an array
let fruits = ["apple", "banana", "cherry"];
console.log(fruits); // ["apple", "banana", "cherry"]
1. push()

Adds a new element at the end of the array.


fruits.push("orange"); // adds "orange" to the end
console.log(fruits); // ["apple", "banana", "cherry", "orange"]
2. pop()

Removes the last element from the array.


fruits.pop(); // removes "orange"
console.log(fruits); // ["apple", "banana", "cherry"]
3. shift()

Removes the first element of the array.


fruits.shift(); // removes "apple"
console.log(fruits); // ["banana", "cherry"]
4. unshift()

Adds a new element at the beginning of the array.


fruits.unshift("kiwi"); // adds "kiwi" at the start
console.log(fruits); // ["kiwi", "banana", "cherry"]
5. includes()

Checks if an array contains a certain value.


console.log(fruits.includes("banana")); // true
console.log(fruits.includes("grape"));  // false
6. indexOf()

Returns the index of a value. If not found, returns -1.


console.log(fruits.indexOf("cherry")); // 2
console.log(fruits.indexOf("grape"));  // -1
7. join()

Joins all array elements into a string, separated by a comma (or custom separator).


console.log(fruits.join());       // "kiwi,banana,cherry"
console.log(fruits.join(" - "));  // "kiwi - banana - cherry"
8. slice()

Returns a portion of the array (non-destructive).


let newFruits = fruits.slice(1, 3); // from index 1 to 2
console.log(newFruits); // ["banana", "cherry"]
9. splice()

Adds/removes items at a specific index (destructive).


fruits.splice(1, 1, "grape"); // removes 1 item at index 1, adds "grape"
console.log(fruits); // ["kiwi", "grape", "cherry"]
10. forEach()

Loops through the array without returning anything.


fruits.forEach(function(fruit) {
  console.log(fruit);
});
11. map()

Creates a new array by applying a function to each element.


let upperFruits = fruits.map(f => f.toUpperCase());
console.log(upperFruits); // ["KIWI", "GRAPE", "CHERRY"]
12. filter()

Returns a new array with elements that match a condition.


let shortNames = fruits.filter(f => f.length <= 5);
console.log(shortNames); // ["kiwi", "grape"]
13. reduce()

Reduces an array to a single value (e.g., sum).


let numbers = [1, 2, 3, 4];
let total = numbers.reduce((sum, num) => sum + num, 0);
console.log(total); // 10

5. Conditionals


let age = 18;

if (age >= 18) {
  console.log("Adult");
} else {
  console.log("Minor");
}

let result = age >= 18 ? "Adult" : "Minor";

6. Loops


// for loop
for (let i = 0; i < 5; i++) {
  console.log(i);
}

// while loop
let i = 0;
while (i < 5) {
  console.log(i);
  i++;
}

// do...while loop
let j = 0;
do {
  console.log(j);
  j++;
} while (j < 5);

// for...of loop
let colors = ["red", "green"];
for (let color of colors) {
  console.log(color);
}

// forEach
colors.forEach(function(c) {
  console.log(c);
});

// break & continue
for (let i = 0; i < 10; i++) {
  if (i === 5) break;
  if (i === 2) continue;
  console.log(i);
}

7. DOM Manipulation

The DOM (Document Object Model) allows JavaScript to interact with HTML elements.

1. Selecting Elements

document.getElementById("myId");
document.querySelector(".myClass");
document.querySelectorAll("p");
2. Changing Content

document.getElementById("title").textContent = "New Title";
document.querySelector("#msg").innerHTML = "<b>Hello</b>";
3. Changing Styles

document.body.style.backgroundColor = "lightblue";
4. Events

document.getElementById("btn").addEventListener("click", function() {
  alert("Button clicked!");
});
5. Creating Elements

let newItem = document.createElement("li");
newItem.textContent = "New Item";
document.querySelector("ul").appendChild(newItem);

8. JSON (JavaScript Object Notation)

JSON is a lightweight format for storing and transporting data, often used when data is sent from a server to a web page.

1. JSON Format

It is text written in JavaScript object syntax but used as data.


{
  "name": "Alice",
  "age": 25,
  "isStudent": true,
  "skills": ["HTML", "CSS", "JS"]
}
2. JavaScript Object vs JSON

JavaScript objects can have methods/functions â€” JSON cannot.


// JavaScript Object
let user = {
  name: "Bob",
  age: 30,
  greet: function() {
    console.log("Hello");
  }
};

// JSON - no functions allowed
let jsonString = '{"name": "Bob", "age": 30}';
3. JSON.stringify()

Converts a JavaScript object into a JSON string.


let obj = { name: "John", age: 22 };
let json = JSON.stringify(obj);
console.log(json); // '{"name":"John","age":22}'
4. JSON.parse()

Converts a JSON string into a JavaScript object.


let jsonString = '{"name":"Jane","age":28}';
let obj = JSON.parse(jsonString);
console.log(obj.name); // "Jane"
